get_loc_10x <- function(loc, barcode, num) {
    row <- loc[barcode,]$row
    col <- loc[barcode,]$col

    col_list <- list()
    for (i in 1:(num + 1)) {
    row_range <-c((row - (i - 1)):(row + (i - 1))) 
    col_range <- c((col - 2 * num + (i - 1)):(col + 2 * num - (i - 1)))
    part_col <- loc[loc$row %in% row_range & loc$col %in% col_range,]
    col_list[[i]] <- rownames(part_col)
    }
    col_list <- unique(unlist(col_list))

}
